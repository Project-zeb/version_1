### OPTION 3: Complete Safe Deployment Script

**Run this all at once in PowerShell:**

```powershell
$projectPath = "c:\Users\Ayaan\Desktop\Project-Zeb\versions_history\projectz_v.2 4\projectz_v.2 4"
cd $projectPath

Write-Host "========================================" -ForegroundColor Cyan
Write-Host "Project-Zeb v2.4 Deployment to Main" -ForegroundColor Cyan
Write-Host "========================================" -ForegroundColor Cyan

# Step 1: Git configuration
Write-Host "`n[STEP 1] Configuring git user..." -ForegroundColor Yellow
git config user.email "your-email@example.com"
git config user.name "Your Name"
Write-Host "✓ Git configured" -ForegroundColor Green

# Step 2: Status check
Write-Host "`n[STEP 2] Checking status..." -ForegroundColor Yellow
$status = git status --porcelain
if ($status) {
    Write-Host "Uncommitted changes found:" -ForegroundColor Yellow
    Write-Host $status
    
    # Step 3: Stage changes
    Write-Host "`n[STEP 3] Staging changes..." -ForegroundColor Yellow
    git add -A
    Write-Host "✓ Changes staged" -ForegroundColor Green
    
    # Step 4: Commit
    Write-Host "`n[STEP 4] Creating commit..." -ForegroundColor Yellow
    git commit -m "v2.4: Production release - security hardened, ready for deployment"
    Write-Host "✓ Commit created" -ForegroundColor Green
} else {
    Write-Host "✓ No uncommitted changes" -ForegroundColor Green
}

# Step 5: Create/switch to main
Write-Host "`n[STEP 5] Switching to main branch..." -ForegroundColor Yellow
git checkout -b main 2>$null
if ($LASTEXITCODE -ne 0) {
    git checkout main
}
Write-Host "✓ On main branch" -ForegroundColor Green

# Step 6: Show commits
Write-Host "`n[STEP 6] Recent commits:" -ForegroundColor Yellow
git log --oneline -5

# Step 7: Verify security files
Write-Host "`n[STEP 7] Security verification:" -ForegroundColor Yellow
if (Test-Path ".gitignore") {
    Write-Host "✓ .gitignore present" -ForegroundColor Green
}
if (Test-Path ".env.example") {
    Write-Host "✓ .env.example present" -ForegroundColor Green
}
if (-not (Test-Path "projectz_v.2\.env")) {
    Write-Host "✓ .env excluded from git" -ForegroundColor Green
} else {
    Write-Host "⚠ WARNING: .env file may need to be removed from tracking" -ForegroundColor Yellow
}

# Step 8: Show remote
Write-Host "`n[STEP 8] Remote configuration:" -ForegroundColor Yellow
git remote -v

# Step 9: Ready to push
Write-Host "`n========================================" -ForegroundColor Green
Write-Host "✓ READY TO PUSH TO GITHUB" -ForegroundColor Green
Write-Host "========================================" -ForegroundColor Green
Write-Host "`nRun this command to push:" -ForegroundColor Cyan
Write-Host "git push origin main --force" -ForegroundColor White

Write-Host "`nOr merge into existing main:" -ForegroundColor Cyan
Write-Host "git push origin HEAD:main" -ForegroundColor White
```

## IF PUSH FAILS

### Issue: "Permission denied"
```powershell
# Setup personal access token or SSH
git remote set-url origin https://YOUR_USERNAME:YOUR_TOKEN@github.com/YOUR_ORG/YOUR_REPO.git
```

### Issue: "Authentication failed"
```powershell
# Use HTTPS instead of SSH
git remote set-url origin https://github.com/YOUR_ORG/YOUR_REPO.git
```

### Issue: "Branch already exists"
```powershell
# Force push (use with caution!)
git push origin main --force
```
